# 时间旅行查询（Time Travel Query）
## 语法
### 方法一：使用`AS OF`语法（推荐）
```sql
-- 查询特定时间点的数据
SELECT * FROM hudi_table 
FOR SYSTEM_TIME AS OF TIMESTAMP '2024-01-01 10:00:00';

-- 或简写为
SELECT * FROM hudi_table 
TIMESTAMP AS OF '2024-01-01 10:00:00';

-- 使用时间戳
SELECT * FROM hudi_table 
TIMESTAMP AS OF '20240101000000';  -- Hudi 的时间戳格式
```

### 方法二：使用表配置
```sql
-- 在查询时指定时间点
SELECT * FROM hudi_table 
/*+ OPTIONS('as.of.timestamp'='2024-01-01 10:00:00') */;

-- 或在创建表时指定
CREATE VIEW hudi_table_as_of AS
SELECT * FROM hudi_table 
/*+ OPTIONS('as.of.timestamp'='2024-01-01 10:00:00') */;
```

### 方法三：使用`HoodieTimeline`API
```sql
-- 查询某个提交之后的数据
SELECT * FROM hudi_table 
/*+ OPTIONS('read.start-commit'='20240101000000') */;

-- 查询某个提交之前的数据
SELECT * FROM hudi_table 
/*+ OPTIONS('read.end-commit'='20240101000000') */;

-- 查询两个提交之间的数据
SELECT * FROM hudi_table 
/*+ OPTIONS(
    'read.start-commit'='20240101000000',
    'read.end-commit'='20240101010000'
) */;
```

## 案例
场景：查询用户在不同时间点的状态

```sql
-- 1. 查询今天凌晨的用户数据
SELECT user_id, name, age, address 
FROM user_profile 
TIMESTAMP AS OF DATE_FORMAT(CURRENT_DATE, 'yyyy-MM-dd 00:00:00');

-- 2. 查询上周一的用户数据
SELECT user_id, name, age, address
FROM user_profile
TIMESTAMP AS OF DATE_SUB(CURRENT_DATE, INTERVAL 7 DAY);

-- 3. 查询特定业务时间点的数据
SELECT 
    user_id,
    name,
    _hoodie_commit_time,
    _hoodie_record_key
FROM user_profile
TIMESTAMP AS OF '2024-01-15 09:30:00'  -- 业务活动开始时间
WHERE user_id IN ('user001', 'user002', 'user003');
```

## 高级时间旅行模式
### 数据对比：当前 vs 历史
```sql
-- 对比当前数据和历史数据
WITH current_data AS (
    SELECT user_id, name, age, address
    FROM user_profile
    WHERE user_id = 'user_001'
),
historical_data AS (
    SELECT user_id, name, age, address
    FROM user_profile 
    TIMESTAMP AS OF '2024-01-01 00:00:00'
    WHERE user_id = 'user_001'
)
SELECT 
    c.user_id,
    c.name as current_name,
    h.name as historical_name,
    c.age as current_age,
    h.age as historical_age,
    c.address as current_address,
    h.address as historical_address,
    CASE 
        WHEN c.name != h.name THEN 'Name changed'
        WHEN c.age != h.age THEN 'Age changed'
        WHEN c.address != h.address THEN 'Address changed'
        ELSE 'No change'
    END as change_type
FROM current_data c
FULL OUTER JOIN historical_data h ON c.user_id = h.user_id;
```

### 趋势分析
```sql
-- 分析用户年龄变化趋势
WITH age_changes AS (
    SELECT 
        user_id,
        _hoodie_commit_time as change_time,
        age,
        LAG(age) OVER (PARTITION BY user_id ORDER BY _hoodie_commit_time) as prev_age
    FROM user_profile
    WHERE user_id = 'user_001'
)
SELECT 
    DATE_FORMAT(FROM_UNIXTIME(CAST(change_time AS BIGINT) / 1000), 'yyyy-MM-dd') as change_date,
    age,
    prev_age,
    age - prev_age as age_change
FROM age_changes
WHERE prev_age IS NOT NULL
  AND age != prev_age
ORDER BY change_date;
```

## 限制和注意事项
### 时间旅行查询的限制
1. 数据保留时间：只能查询保留期内的历史数据
    - 受 `clean.policy` 和 `archive.*` 配置限制
2. 查询性能：历史查询可能比当前查询慢
    - 因为需要读取多个版本的文件
3. MOR表 vs COW表：
    - MOR表：可以查询任意时间点，包括未压缩的 log文件
    - COW表：只能查询已提交的快照
4. 资源消耗：历史查询可能消耗更多内存和 CPU

### 推荐配置
```sql
-- 对于需要频繁时间旅行的表
CREATE TABLE ... WITH (
    -- 保留足够的历史
    'cleaner.commits.retained' = '100',
    'archive.min_commits' = '50',
    'archive.max_commits' = '100',
    
    -- 启用元数据加速
    'hoodie.metadata.enable' = 'true',
    'hoodie.metadata.index.column.stats.enable' = 'true',
    
    -- 对于MOR表，控制log文件大小
    'hoodie.logfile.max.size' = '134217728',  -- 128MB
    'compaction.delta_commits' = '5'  -- 每5次提交压缩一次
);
```

## 实用技巧与实践
### 创建时间旅行视图
```sql
-- 创建常用时间点的视图
CREATE VIEW user_profile_yesterday AS
SELECT * FROM user_profile 
TIMESTAMP AS OF DATE_FORMAT(DATE_SUB(CURRENT_DATE, INTERVAL 1 DAY), 'yyyy-MM-dd 23:59:59');

CREATE VIEW user_profile_last_week AS
SELECT * FROM user_profile 
TIMESTAMP AS OF DATE_FORMAT(DATE_SUB(CURRENT_DATE, INTERVAL 7 DAY), 'yyyy-MM-dd 23:59:59');

CREATE VIEW user_profile_month_start AS
SELECT * FROM user_profile 
TIMESTAMP AS OF DATE_FORMAT(DATE_TRUNC('MONTH', CURRENT_DATE), 'yyyy-MM-dd 00:00:00');
```

### 自动化时间点管理
```sql
-- 使用变量管理时间点
SET 'query.timestamp' = DATE_FORMAT(DATE_SUB(CURRENT_DATE, INTERVAL 1 DAY), 'yyyy-MM-dd 23:59:59');

SELECT * FROM user_profile 
TIMESTAMP AS OF '${query.timestamp}';

-- 或在Flink SQL客户端中
EXECUTE STATEMENT SET 
BEGIN
    INSERT INTO report_table
    SELECT * FROM user_profile 
    TIMESTAMP AS OF '${query.timestamp}';
END;
```

# <font style="color:rgb(15, 17, 21);">查询示例</font>
## 查询数据的历史及变化
查询某条记录的所有版本

```sql
-- 查看某条记录的所有版本
SELECT 
    _hoodie_commit_time,
    _hoodie_operation,
    _hoodie_file_name,
    user_id,
    name,
    age,
    address
FROM user_profile
WHERE _hoodie_record_key = 'user_001'
ORDER BY _hoodie_commit_time DESC;
```



数据变化追踪

```sql
-- 追踪某个字段的历史变化
WITH user_history AS (
    SELECT 
        _hoodie_commit_time as change_time,
        name,
        age,
        address,
        LAG(name) OVER (ORDER BY _hoodie_commit_time) as prev_name,
        LAG(age) OVER (ORDER BY _hoodie_commit_time) as prev_age,
        LAG(address) OVER (ORDER BY _hoodie_commit_time) as prev_address
    FROM user_profile
    WHERE _hoodie_record_key = 'user_001'
)
SELECT 
    change_time,
    CASE 
        WHEN name != prev_name THEN CONCAT('Name changed: ', prev_name, ' → ', name)
        WHEN age != prev_age THEN CONCAT('Age changed: ', CAST(prev_age AS STRING), ' → ', CAST(age AS STRING))
        WHEN address != prev_address THEN CONCAT('Address changed: ', prev_address, ' → ', address)
        ELSE 'No change'
    END as change_description
FROM user_history
WHERE name != prev_name 
   OR age != prev_age 
   OR address != prev_address;
```

## 数据质量检查
```sql
-- 检查数据分布
SELECT 
    DATE_FORMAT(FROM_UNIXTIME(CAST(_hoodie_commit_time AS BIGINT) / 1000), 'yyyy-MM-dd HH') as commit_hour,
    _hoodie_operation,
    COUNT(*) as record_count,
    COUNT(DISTINCT _hoodie_record_key) as unique_keys,
    COUNT(DISTINCT _hoodie_partition_path) as partitions_updated
FROM user_profile
WHERE _hoodie_commit_time >= '20240101000000'
GROUP BY 1, 2
ORDER BY 1 DESC, 2;
```

## 增量查询（Incremental Query）
### 基础增量查询
```sql
-- 查询某个时间点之后的变化
SELECT * FROM user_profile 
/*+ OPTIONS('read.start-commit'='20240101000000') */;

-- 查询某个时间范围内的变化
SELECT * FROM user_profile 
/*+ OPTIONS(
    'read.start-commit'='20240101000000',
    'read.end-commit'='20240102000000'
) */;

-- 结合系统字段过滤
SELECT 
    _hoodie_commit_time,
    _hoodie_operation,
    user_id,
    name,
    age
FROM user_profile 
/*+ OPTIONS('read.start-commit'='20240101000000') */
WHERE _hoodie_operation IN ('INSERT', 'UPDATE');
```

### 增量数据同步到其他系统
```sql
-- 将增量数据写入Kafka
INSERT INTO kafka_changelog
SELECT 
    _hoodie_commit_time as change_time,
    _hoodie_record_key as record_key,
    _hoodie_operation as operation,
    user_id,
    name,
    age,
    address,
    CURRENT_TIMESTAMP as sync_time
FROM user_profile 
/*+ OPTIONS('read.start-commit'='${last_sync_time}') */
WHERE _hoodie_operation != 'DELETE';  -- 排除删除操作
```

# FlinkSQL 配置优化
## 创建表时的配置
```sql
CREATE TABLE user_profile (
    user_id STRING,
    name STRING,
    age INT,
    address STRING,
    create_time TIMESTAMP(3),
    PRIMARY KEY (user_id) NOT ENFORCED
) WITH (
    'connector' = 'hudi',
    'path' = 'hdfs:///user/hudi/user_profile',
    'table.type' = 'MERGE_ON_READ',
    
    -- 时间旅行相关配置
    'read.tasks' = '4',  -- 并行读取任务数
    'read.streaming.enabled' = 'true',  -- 启用流式读取
    'read.streaming.skip_compaction' = 'true',  -- 跳过压缩文件
    
    -- 快照版本保留策略
    'cleaner.policy' = 'KEEP_LATEST_COMMITS',
    'cleaner.commits.retained' = '10',  -- 保留最近10次提交
    
    -- 用于时间旅行的元数据保留
    'archive.min_commits' = '20',  -- 最少保留20个提交
    'archive.max_commits' = '30',  -- 最多保留30个提交
    
    -- 查询优化
    'hoodie.metadata.enable' = 'true'  -- 启用元数据表加速查询
);
```
